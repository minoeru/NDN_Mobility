on run	-- run nlsr	repeat with i from 1 to <max_num>		my newTab("docker exec -it nlsr" & i & " /bin/bash", i)	end repeat		-- Router1	my newTab2("docker exec -it nlsr1 /bin/bash -c \"nfdc cs config serve off\"")	--PC	my terminalCommand("docker exec -it nlsr11 /bin/bash -c \"nfdc cs config serve off\"")		-- Center_Router	repeat with i from <center_num> to <center_num2>		my newTab2("docker exec -it nlsr" & i & " /bin/bash -c \"/NLSR/routing.sh 10 0\"")		-- moveの広告確率 その他の広告確率	end repeat		-- Edge_Router	repeat with i from <edge_num> to <edge_num2>		my newTab2("docker exec -it nlsr" & i & " /bin/bash -c \"/NLSR/routing.sh 10 0\"")		-- moveの広告確率 その他の広告確率	end repeat		-- Server	repeat with i from 1 to 4		my newTab2("docker exec -it nlsr" & i + 11 & " /bin/bash")		my terminalCommand("./doPoke.sh list" & i & ".txt ")	end repeat		tell application "System Events"		keystroke "m" using command down	end tell	end runon newTab(command, num)	-- open new terminal	tell application "System Events"		keystroke "t" using command down	end tell	delay 1.0	-- start nfd		tell application "Terminal"		do script command in front window		delay 0.5		do script "sed -i -e 's/default_level.INFO/default_level DEBUG/g'  /usr/local/etc/ndn/nfd.conf" in front window		do script "nfd-start 2> nfd-log.txt" in front window	end tell	delay 1.0	-- nfd to background	tell application "System Events"		keystroke "c" using control down	end tell	delay 1.0	-- start NLSR	tell application "Terminal"
		<processing>		delay 0.5	end tellend newTabon newTab2(command)	-- open new terminal	tell application "System Events"		keystroke "t" using command down		delay 0.5	end tell	terminalCommand(command)end newTab2on terminalCommand(command)	tell application "Terminal"		do script command in front window	end tell	delay 0.5end terminalCommand
